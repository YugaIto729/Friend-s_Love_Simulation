VALUE_SET 121 0
DAILOG_WAIT 1 "[121] != 0"
DIALOG_REMOVE 1
BRANCH "[121] == 1"
	MESS "はいを選択"
ENDBRANCH
BRANCH "[121] == 2"
	MESS "いいえを選択"
ENDBRANCH

MESS "ダイアログ終了"
#EVENTEND

#============================================
#ダイアログを使う時はルールを厳守
#============================================

 	MESS "YES"
ELSE_DIALOG

	MESS "NO"
END_DIALOG
